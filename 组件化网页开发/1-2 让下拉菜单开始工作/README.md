     }

    function dropdown(elem) {
        var $elem = $(elem),

        $elem.hover(function() {
            $elem.addClass(activeClass);
        }, function() {
            $elem.removeClass(activeClass);
        });

     }

    // 插件的使用方法
    $.fn.extend({
      dropdown:function(){
            // return this;    // 就是下面的$('.dropdown')  可能是个数组哦
        return this.each(function(){
          dropdown(this);

        });

      }
    });

    $('.dropdown').dropdown(); // 插件为了更好的调用

    'use strict';

    function dropdown(elem) {

            $elem.addClass(activeClass);
        }, function() {
            $elem.removeClass(activeClass);
        });

     };

    $.fn.extend({

    });
